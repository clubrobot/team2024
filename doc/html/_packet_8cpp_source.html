53/*
54 uint8_t Packet::constructPacket(const uint16_t& messageLen, const uint8_t& packetID)
55 Description:
56 ------------
57 * Calculate, format, and insert the packet protocol metadata into the packet transmit
58 buffer
59 Inputs:
60 -------
61 * const uint16_t& messageLen - Number of values in txBuff
62 to send as the payload in the next packet
63 * const uint8_t& packetID - The packet 8-bit identifier
64 Return:
65 -------
66 * uint8_t - Number of payload bytes included in packet
67*/
68uint8_t Packet::constructPacket(const uint16_t& messageLen, const uint8_t& packetID)
69{
70 if (messageLen > MAX_PACKET_SIZE)
71 {
72 calcOverhead(txBuff, MAX_PACKET_SIZE);
73 stuffPacket(txBuff, MAX_PACKET_SIZE);
74 uint8_t crcVal = crc.calculate(txBuff, MAX_PACKET_SIZE);
75
76 preamble[1] = packetID;
77 preamble[2] = overheadByte;
78 preamble[3] = MAX_PACKET_SIZE;
79
80 postamble[0] = crcVal;
81
82 return MAX_PACKET_SIZE;
83 }
84 else
85 {
86 calcOverhead(txBuff, (uint8_t)messageLen);
87 stuffPacket(txBuff, (uint8_t)messageLen);
88 uint8_t crcVal = crc.calculate(txBuff, (uint8_t)messageLen);
89
90 preamble[1] = packetID;
91 preamble[2] = overheadByte;
92 preamble[3] = messageLen;
93
94 postamble[0] = crcVal;
95
96 return (uint8_t)messageLen;
97 }
98}

102 uint8_t Packet::parse(const uint8_t& recChar, const bool& valid)
103 Description:
104 ------------
105 * Parses incoming serial data, analyzes packet contents,
106 and reports errors/successful packet reception. Executes
107 callback functions for parsed packets whos ID has a
108 corresponding callback function set via
109 "void Packet::begin(const configST configs)"
110 Inputs:
111 -------
112 * const uint8_t& recChar - Next char to parse in the stream
113 * const bool& valid - Set if stream is "available()" and clear if not
114 Return:
115 -------
116 * uint8_t - Num bytes in RX buffer
117*/
118
119uint8_t Packet::parse(const uint8_t& recChar, const bool& valid)
120{
121 bool packet_fresh = (packetStart == 0) || ((millis() - packetStart) < timeout);
122
123 if(!packet_fresh) //packet is stale, start over.
124 {
125 if (debug)
126 debugPort->println("ERROR: STALE PACKET");
127
128 bytesRead = 0;
129 state = find_start_byte;
130 status = STALE_PACKET_ERROR;
131 packetStart = 0;
132
133 return bytesRead;
134 }
135
136 if (valid)
137 {
138
139 switch (state)
140 {
141 case find_start_byte:
142 {
143 if (recChar == START_BYTE)
144 {
145 state = find_id_byte;
146 packetStart = millis(); //start the timer
147 }
148
149 break;
150 }
151
152 case find_id_byte:
153 {
154 idByte = recChar;
155 state = find_overhead_byte;
156 break;
157 }
158
159 case find_overhead_byte:
160 {
161 recOverheadByte = recChar;
162 state = find_payload_len;
163 break;
164 }
165
166 case find_payload_len:
167 {
168 if ((recChar > 0) && (recChar <= MAX_PACKET_SIZE))
169 {
170 bytesToRec = recChar;
171 payIndex = 0;
172 state = find_payload;
173 }
174 else
175 {
176 bytesRead = 0;
177 state = find_start_byte;
178 status = PAYLOAD_ERROR;
179
180 if (debug)
181 debugPort->println("ERROR: PAYLOAD_ERROR");
182
183 reset();
184 return bytesRead;
185 }
186 break;
187 }
188
189 case find_payload:
190 {
191 if (payIndex < bytesToRec)
192 {
193 rxBuff[payIndex] = recChar;
194 payIndex++;
195
196 if (payIndex == bytesToRec)
197 state = find_crc;
198 }
199 break;
200 }
201
202 case find_crc:
203 {
204 uint8_t calcCrc = crc.calculate(rxBuff, bytesToRec);
205
206 if (calcCrc == recChar)
207 state = find_end_byte;
208 else
209 {
210 bytesRead = 0;
211 state = find_start_byte;
212 status = CRC_ERROR;
213
214 if (debug)
215 debugPort->println("ERROR: CRC_ERROR");
216
217 reset();
218 return bytesRead;
219 }
220
221 break;
222 }
223
224 case find_end_byte:
225 {
226 state = find_start_byte;
227
228 if (recChar == STOP_BYTE)
229 {
230 unpackPacket(rxBuff);
231 bytesRead = bytesToRec;
232 status = NEW_DATA;
233
234 if (callbacks)
235 {
236 if (idByte < callbacksLen){
237 callbacks[idByte]();
238 }else if (debug)
239 {
240 debugPort->print(F("ERROR: No callback available for packet ID "));
241 debugPort->println(idByte);
242 }
243 }
244 packetStart = 0; // reset the timer
245 return bytesToRec;
246 }
247
248 bytesRead = 0;
249 status = STOP_BYTE_ERROR;
250
251 if (debug)
252 debugPort->println("ERROR: STOP_BYTE_ERROR");
253
254 reset();
255 return bytesRead;
256 break;
257 }
258
259 default:
260 {
261 if (debug)
262 {
263 debugPort->print("ERROR: Undefined state ");
264 debugPort->println(state);
265 }
266
267 reset();
268 bytesRead = 0;
269 state = find_start_byte;
270 break;
271 }
272 }
273 }
274 else
275 {
276 bytesRead = 0;
277 status = NO_DATA;
278 return bytesRead;
279 }
280
281 bytesRead = 0;
282 status = CONTINUE;
283 return bytesRead;
284}

306 void Packet::calcOverhead(uint8_t arr[], const uint8_t &len)
307 Description:
308 ------------
309 * Calculates the COBS (Consistent Overhead Stuffing) Overhead
310 byte and stores it in the class's overheadByte variable. This
311 variable holds the byte position (within the payload) of the
312 first payload byte equal to that of START_BYTE
313 Inputs:
314 -------
315 * uint8_t arr[] - Array of values the overhead is to be calculated
316 over
317 * const uint8_t &len - Number of elements in arr[]
318 Return:
319 -------
320 * void
321*/
322void Packet::calcOverhead(uint8_t arr[], const uint8_t& len)
323{
324 overheadByte = 0xFF;
325
326 for (uint8_t i = 0; i < len; i++)
327 {
328 if (arr[i] == START_BYTE)
329 {
330 overheadByte = i;
331 break;
332 }
333 }

338 int16_t Packet::findLast(uint8_t arr[], const uint8_t &len)
339 Description:
340 ------------
341 * Finds last instance of the value START_BYTE within the given
342 packet array
343 Inputs:
344 -------
345 * uint8_t arr[] - Packet array

348 -------

352int16_t Packet::findLast(uint8_t arr[], const uint8_t& len)
353{
354 for (uint8_t i = (len - 1); i != 0xFF; i--)

362/*
363 void Packet::stuffPacket(uint8_t arr[], const uint8_t &len)
364 Description:
365 ------------
366 * Enforces the COBS (Consistent Overhead Stuffing) ruleset across
367 all bytes in the packet against the value of START_BYTE
368 Inputs:
369 -------
370 * uint8_t arr[] - Array of values to stuff
371 * const uint8_t &len - Number of elements in arr[]
372 Return:
373 -------
374 * void
375*/
376void Packet::stuffPacket(uint8_t arr[], const uint8_t& len)
377{
378 int16_t refByte = findLast(arr, len);
379
380 if (refByte != -1)
381 {
382 for (uint8_t i = (len - 1); i != 0xFF; i--)
383 {
384 if (arr[i] == START_BYTE)
385 {
386 arr[i] = refByte - i;
387 refByte = i;
388 }
389 }
390 }
391}

394/*
395 void Packet::unpackPacket(uint8_t arr[], const uint8_t &len)
396 Description:
397 ------------
398 * Unpacks all COBS-stuffed bytes within the array
399 Inputs:
400 -------
401 * uint8_t arr[] - Array of values to unpack
402 * const uint8_t &len - Number of elements in arr[]
403 Return:
404 -------
405 * void
406*/
407void Packet::unpackPacket(uint8_t arr[])
408{
409 uint8_t testIndex = recOverheadByte;
410 uint8_t delta = 0;
411
412 if (testIndex <= MAX_PACKET_SIZE)
413 {
414 while (arr[testIndex])
415 {
416 delta = arr[testIndex];
417 arr[testIndex] = START_BYTE;
418 testIndex += delta;
419 }
420 arr[testIndex] = START_BYTE;
421 }
422}
