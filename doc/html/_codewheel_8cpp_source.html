40void Codewheel::update()
41{
42 digitalWrite(m_COUNTER_XY, m_COUNTER_AXIS);
43 digitalWrite(m_COUNTER_OE, LOW);
44 m_currentCounter = 0;
45 for (int i = 0; i < 4; i++)
46 {
47 // Select one byte of the 32 bits quad counter
48 // Its value will be stored in the shift register
49 byte SEL1 = i & 0x01;
50 byte SEL2 = (~i >> 1) & 0x01;
51 digitalWrite(m_COUNTER_SEL1, SEL1);
52 digitalWrite(m_COUNTER_SEL2, SEL2);
53
54 // The counter value is received in big endian order
55 m_currentCounter <<= 8;
56
57 // Read the value stored in the shift register
58 digitalWrite(m_REGISTER_CLOCK, HIGH);
59 digitalWrite(m_REGISTER_LATCH, HIGH);
60 m_currentCounter += shiftIn(m_REGISTER_DATA, m_REGISTER_CLOCK, MSBFIRST);
61 digitalWrite(m_REGISTER_LATCH, LOW);
62 }
63 digitalWrite(m_COUNTER_OE, HIGH);
64}
