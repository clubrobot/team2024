38bool PurePursuit::checkLookAheadGoal(const float x, const float y)
39{
40 // The purpose of this function is to find the intersection point between a circle of radius
41 // `m_lookAhead` centered at the robot position and the path. As there may be several of them,
42 // we iterate through the path segments in the order of passing and stop as soon as we find
43 // one.
44 for (int i = m_goalIndex; i < m_numWaypoints; i++)
45 {
46 float dx = x - m_waypoints[i].x;
47 float dy = y - m_waypoints[i].y;
48 float edgedx, edgedy;
49 if (i < m_numWaypoints-1)
50 {
51 edgedx = m_waypoints[i+1].x - m_waypoints[i].x;
52 edgedy = m_waypoints[i+1].y - m_waypoints[i].y;
53 }
54 else
55 {
56 edgedx = cos(m_finalAngle);
57 edgedy = sin(m_finalAngle);
58 }
59 float edgeLength = sqrt(edgedx * edgedx + edgedy * edgedy);
60
61 // `h` is the distance between the robot and the current line (i.e. the current segment but
62 // without regard to its endpoints).
63 float h = fabs(edgedx * dy - edgedy * dx) / edgeLength;
64 if (m_lookAhead < h) // There is no intersection between the current line and the circle.
65 continue;
66
67 // `t` is the relative position of the robot projection in the current segment.
68 // `t = 0` means that the robot projection is on its first endpoint and `t = 1` means
69 // that the robot projection is on its second endpoint.
70 float t = (edgedx * dx + edgedy * dy) / (edgeLength * edgeLength);
71
72 // There is two intersection points between the circle and the line but we only consider
73 // the one ahead of the robot: if it is beyond the second endpoint (so it's not on the
74 // segment), then we go to the next.
75 float t1 = t - sqrt(m_lookAhead * m_lookAhead - h * h) / edgeLength;
76 float t2 = t + sqrt(m_lookAhead * m_lookAhead - h * h) / edgeLength;
77
78 // Skip if the intersection point is beyond the second endpoint (see above).
79 if (t2 < 0)
80 continue;
81 else if (t2 > 1 && i < m_numWaypoints-1)
82 continue;
83 else if (t1 > 1 && i == m_numWaypoints-1)
84 continue;
85 else if (t2 > 1 + m_lookAheadBis / edgeLength)
86 t2 = 1 + m_lookAheadBis / edgeLength;
87
88 // Save the new goal.
89 if (i > m_goalIndex || t2 > m_goalParam)
90 {
91 m_goalIndex = i;
92 m_goalParam = t2;
93 }
94 return true;
95 }
96 return false;
97}

99void PurePursuit::checkProjectionGoal(const float x, const float y)
100{
101 // The purpose of this function is to find the closest point between the path and the robot. To
102 // do so we iterate through all the segments, cmpute their distance to the robot and sort them.
103 float hmin = INFINITY;
104 for (int i = m_goalIndex; i < m_numWaypoints-1; i++)
105 {
106 float dx = x - m_waypoints[i].x;
107 float dy = y - m_waypoints[i].y;
108 float edgedx = m_waypoints[i+1].x - m_waypoints[i].x;
109 float edgedy = m_waypoints[i+1].y - m_waypoints[i].y;
110 float edgeLength = sqrt(edgedx * edgedx + edgedy * edgedy);
111
112 // `t` and `h` have the same meaning than in the `checkLookAheadGoal` method.
113 float h, t = (edgedx * dx + edgedy * dy) / (edgeLength * edgeLength);
114 if (t > 1 && i+1 < m_numWaypoints-1)
115 continue;
116
117 if (t > 1) // The closest point of the segment is its second endpoint.
118 {
119 float dx2 = x - m_waypoints[i+1].x;
120 float dy2 = y - m_waypoints[i+1].y;
121 h = sqrt(dx2 * dx2 + dy2 * dy2);
122 t = 1;
123 }
124 else if (t <= 0) // The closest point of the segment is its first endpoint.
125 {
126 h = sqrt(dx * dx + dy * dy);
127 t = 0;
128 }
129 else
130 {
131 h = fabs(edgedx * dy - edgedy * dx) / edgeLength;
132 }
133
134 // Save the new projection point if it is closer than the current one.
135 if (h < hmin)
136 {
137 hmin = h;
138 if (i > m_goalIndex || t > m_goalParam)
139 {
140 m_goalIndex = i;
141 m_goalParam = t;
142 }
143 }
144 }
145}

147float PurePursuit::getDistAfterGoal()
148{
149 // This function computes the remaining distance between the current goal and the last
150 // waypoint. It basically does the sum of all the last segments lengths.
151 float dist = 0;
152 for (int i = m_goalIndex; i < m_numWaypoints; i++)
153 {
154 float edgedx, edgedy;
155 if (i < m_numWaypoints - 1)
156 {
157 edgedx = m_waypoints[i+1].x - m_waypoints[i].x;
158 edgedy = m_waypoints[i+1].y - m_waypoints[i].y;
159 }
160 else
161 {
162 edgedx = cos(m_finalAngle);
163 edgedy = sin(m_finalAngle);
164 }
165 float edgeLength = sqrt(edgedx * edgedx + edgedy * edgedy);
166 if (i == m_goalIndex)
167 dist += (1 - m_goalParam) * edgeLength;
168 else
169 dist += edgeLength;
170 }
171 return dist;
172}

174void PurePursuit::computeVelSetpoints(float timestep)
175{
176 // Aliases
177 const float x = getPosInput().x;
178 const float y = getPosInput().y;
179 const float theta = getPosInput().theta;
180 const float linVelKp = getLinVelKp();
181 const float angVelKp = getLinVelKp();
182 const float linVelMax = getLinVelMax();
183 const float angVelMax = getAngVelMax();
184
185 // Compute the goal position on the path: `m_goalIndex` is the index of the segment it is
186 // located on and `m_goalParam` is its relative position on that segment.
187 if (!checkLookAheadGoal(x, y))
188 checkProjectionGoal(x, y);
189
190 // Compute the goal cartesian position.
191 int i = m_goalIndex;
192 float t = m_goalParam;
193 Waypoint goal;
194 if (i < m_numWaypoints - 1)
195 {
196 goal.x = (1-t) * m_waypoints[i].x + t * m_waypoints[i+1].x;
197 goal.y = (1-t) * m_waypoints[i].y + t * m_waypoints[i+1].y;
198 }
199 else
200 {
201 goal.x = m_waypoints[i].x + t * cos(m_finalAngle);
202 goal.y = m_waypoints[i].y + t * sin(m_finalAngle);
203 }
204
205 // Compute the norm and the argument of the vector going from the robot to its goal.
206 float chord = sqrt((goal.x - x) * (goal.x - x) + (goal.y - y) * (goal.y - y));
207 float delta = atan2(goal.y - y, goal.x - x) - theta + M_PI / 2 * (1 - m_direction);
208
209 // The minimum curvature that the robot must follow is `2 * sin(delta) / chord`. So we deduce
210 // from this and the maximum linear velocity allowed the maximum angular velocity setpoint. If
211 // it exceeds the maximum angular velocity allowed, then we do the opposite: we use the maximum
212 // angular velocity allowed to compute the maximum linear velocity setpoint.
213 float newLinVelMax = linVelMax;
214 float newAngVelMax = angVelMax;
215 if (newAngVelMax * chord >= newLinVelMax * fabs(2 * sin(delta)))
216 newAngVelMax = newLinVelMax * fabs(2 * sin(delta)) / chord;
217 else
218 newLinVelMax = newAngVelMax * chord / fabs(2 * sin(delta));
219
220 // Then we do a simple proportional control for both linear and angular velocities.
221 float linPosSetpoint = (chord + getDistAfterGoal()) * m_direction;
222 float linVelSetpoint = saturate(linVelKp * linPosSetpoint, -newLinVelMax, newLinVelMax);
223
224 float angPosSetpoint = inrange(delta, -M_PI, M_PI);
225 float angVelSetpoint = saturate(angVelKp * angPosSetpoint, -newAngVelMax, newAngVelMax);
226
227 // When traveling on the path, we slow down the robot if its orientation is farther than it
228 // should. The following is an empiric but continuous formula.
229 if (cos(delta) > 0)
230 linVelSetpoint *= (1 + cos(2 * delta)) / 2;
231 else
232 linVelSetpoint *= 0;
233
234 // This could be computed elsewhere but here is convenient.
235 m_goalReached = fabs(chord + getDistAfterGoal()) < getLinPosThreshold();
236
237 setVelSetpoints(linVelSetpoint, angVelSetpoint);
238}
