49void GOTO_DELTA()
50{
51 purePursuit.reset();
52 positionControl.disable();
53
54 Position initial_pos = odometry.getPosition();
55
56 float dx = talking.read<float>();
57 float dy = talking.read<float>();
58 talking.endTranfert();
59
60 Position target_pos;
61 target_pos.x = initial_pos.x + dx*cos(initial_pos.theta) + dy*-1*sin(initial_pos.theta);
62 target_pos.y = initial_pos.y + dx*sin(initial_pos.theta) + dy*cos(initial_pos.theta);
63
64 target_pos.theta = atan2(target_pos.y-initial_pos.y,target_pos.x-initial_pos.x);
65 int direction;
66
67 initial_pos.theta = inrange(initial_pos.theta,-M_PI,M_PI);
68
69 if (fabs(inrange(target_pos.theta - initial_pos.theta,-M_PI,M_PI))<(M_PI/2))
70 {
71 direction = PurePursuit::FORWARD;
72 }else{
73 direction = PurePursuit::BACKWARD;
74 }
75
76 purePursuit.setDirection((PurePursuit::Direction) direction);
77 purePursuit.addWaypoint(PurePursuit::Waypoint(initial_pos.x, initial_pos.y));
78 purePursuit.addWaypoint(PurePursuit::Waypoint(target_pos.x, target_pos.y));
79
80 purePursuit.setFinalAngle(target_pos.theta);
81
82 positionControl.setPosSetpoint(Position(target_pos.x, target_pos.y, target_pos.theta + direction * M_PI));
83
84 // Enable PurePursuit controller
85 velocityControl.enable();
86 positionControl.setMoveStrategy(purePursuit);
87 positionControl.enable();
88}

129void START_PUREPURSUIT()
130{
131 // Setup PurePursuit
132 byte direction = talking.read<byte>();
133 switch (direction)
134 {
135 case 0: purePursuit.setDirection(PurePursuit::FORWARD); break;
136 case 1: purePursuit.setDirection(PurePursuit::BACKWARD); break;
137 }
138 purePursuit.setFinalAngle(talking.read<float>());
139 talking.endTranfert();
140 // Compute final setpoint
141 const PurePursuit::Waypoint wp0 = purePursuit.getWaypoint(purePursuit.getNumWaypoints() - 2);
142 const PurePursuit::Waypoint wp1 = purePursuit.getWaypoint(purePursuit.getNumWaypoints() - 1);
143 positionControl.setPosSetpoint(Position(wp1.x, wp1.y, atan2(wp1.y - wp0.y, wp1.x - wp0.x) + direction * M_PI));
144
145 // Enable PurePursuit controller
146 velocityControl.enable();
147 positionControl.setMoveStrategy(purePursuit);
148 positionControl.enable();
149}

160void START_TURNONTHESPOT()
161{
162 Position posSetpoint = odometry.getPosition();
163 float initTheta = posSetpoint.theta;
164 posSetpoint.theta = talking.read<float>();
165 float angPosSetpoint = inrange((posSetpoint.theta - initTheta), -M_PI, M_PI);
166 velocityControl.enable();
167 positionControl.setPosSetpoint(posSetpoint);
168 if(talking.read<byte>()){
169 if(angPosSetpoint>0) turnOnTheSpot.setDirection(TurnOnTheSpot::CLOCK);
170 else turnOnTheSpot.setDirection(TurnOnTheSpot::TRIG);
171 }
172 else{
173 if(angPosSetpoint>0) turnOnTheSpot.setDirection(TurnOnTheSpot::TRIG);
174 else turnOnTheSpot.setDirection(TurnOnTheSpot::CLOCK);
175 }
176 talking.endTranfert();
177 positionControl.setMoveStrategy(turnOnTheSpot);
178 positionControl.enable();
179}
