115void SerialTalking::generateRandomUUID(char* uuid, int length){
116 // Initialize the random number generator
117 randomSeed(analogRead(0));
118
119 // Generate the UUID from a list of random hexadecimal numbers
120 for (int i = 0; i < length; i++)
121 {
122 if (i % 5 == 4)
123 uuid[i] = '-';
124 else
125 {
126 long digit = random(16);
127 if (digit < 10)
128 uuid[i] = char('0' + digit);
129 else
130 uuid[i] = char('a' + digit - 10);
131 }
132 }
133 uuid[length] = '\0';
134}
