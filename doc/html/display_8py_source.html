56 def upload_char_pattern(self, char, pattern):
57 lines = []
58 pattern = pattern.convert('L')
59 for y in range(8):
60 line = 0
61 for x in range(pattern.width):
62 digit = ~(pattern.getpixel((x, y)) // 255)
63 line = (line << 1) | (digit & 1)
64 line <<= 8 - pattern.width
65 lines.append(BYTE(line))
