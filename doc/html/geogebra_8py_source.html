213 command = ElementTree.SubElement(self.construction, "command")

311 def _parse_point(self, element):
312 coords = element.find('coords')
313 x = float(coords.get('x'))
314 y = float(coords.get('y'))
315 z = float(coords.get('z'))
316 return Geogebra.Point((x / z, y / z))

318 def _parse_line(self, element):
319 coords = element.find('coords')
320 a = float(coords.get('x'))
321 b = float(coords.get('y'))
322 c = float(coords.get('z'))
323 try:
324 x, y = -c / a, 0
325 except ZeroDivisionError:
326 x, y = 0, -c / b
327 return Geogebra.Line((x, y, math.pi - math.atan2(a, b)))

329 def _parse_conic(self, element):
330 matrix = element.find('matrix').attrib
331 A = float(matrix.get('A0'))
332 C = float(matrix.get('A1'))
333 F = -float(matrix.get('A2'))
334 B = 2 * float(matrix.get('A3'))
335 D = 2 * float(matrix.get('A4'))
336 E = 2 * float(matrix.get('A5'))
337 if A * C - (B / 2) ** 2 < 0:
338 raise ValueError('not an ellipse')
339 if A == C and B == 0:
340 xc = -D / (2 * A)
341 yc = -E / (2 * C)
342 radius = math.sqrt(xc ** 2 + yc ** 2 + F)
343 return Geogebra.Circle((xc, yc, radius))
344 else:
345 raise NotImplementedError('ellipses currently not handled')

347 def _parse_segment(self, element):
348 label = element.attrib['label']
349 command = self.root.find("./construction/command[@name='Segment']/output[@a0='{}']/..".format(label))

358 i = [output.get(key) for key in sorted(output.attrib.keys())].index(label) - 1

401 def _parse_polygon_from_command(self, command):
402 input = command.find('input')
403 try:
404 numvertices = int(input.get('a2'))
405 output = command.find('output')
406 labels = [input.get('a0'), input.get('a1')] + [output.get(key) for key in
407 sorted(output.attrib.keys())[2 - numvertices:]]
408 except (TypeError, ValueError):
409 labels = [input.get(key) for key in sorted(input.attrib.keys())]
410 return Geogebra.Polygon([self.get(label) for label in labels])
