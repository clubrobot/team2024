351 def stuff_packet(self, pay_len):
352 '''
353 Description:
354 ------------
355 Enforces the COBS (Consistent Overhead Stuffing) ruleset across
356 all bytes in the packet against the value of START_BYTE
357
358 :param pay_len: int - number of bytes in the payload
359
360 :return: void
361 '''
362
363 refByte = self.find_last(pay_len)
