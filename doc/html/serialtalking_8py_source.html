75 def connect(self, timeout=5):
76 '''! Connecte à un arduino
77
78 @param timeout timeout de bas à 5 sec
79 '''
80 if self.is_connected:
81 raise AlreadyConnectedError('{} is already connected'.format(self.port))
82
83 # Connect to the serial port
84 try:
85 self.link = txfer.SerialTransfer(self.port, BAUDRATE)
86 if(self.link.open()==False):
87 raise ConnectionFailedError
88 except Exception as e:
89 raise ConnectionFailedError(str(e))
90
91 startingtime = time.monotonic()
92 while not self.is_connected:
93 try:
94 shouldBePong = self.request(PING_OPCODE, STRING)[0]
95 if(shouldBePong=="pong\x00"):
96 self.is_connected = True
97 time.sleep(0.1)
98 if(time.monotonic() - startingtime > timeout): raise TimeoutError
99 except TimeoutError:
100 if time.monotonic() - startingtime > timeout:
101 self.disconnect()
102 raise MuteError(
103 '\'{}\' is mute. It may not be an Arduino or it\'s sketch may not be correctly loaded.'.format(
104 self.port)) from None
105 else:
106 continue
107
108 except NotConnectedError:
109 self.is_connected = False

160 def order(self, opcode, *args):
161 '''! Envoie un ordre à l'arduino
162 @param opcode Code de l'opération
163 @param *args arguments d'envoi
164 '''
165 if(len(args)==0): args = [BYTE(0x01)] #Si pas d'args
166 #Oui le string fait chier
167 for arg in args:
168 if(arg["format"]=="str"):
169 self.write_buffer(arg, len(arg["value"]))
170 else:
171 self.write_buffer(arg)
172
173 self.send_buffer(opcode)
174 return

177 def request(self, opcode, *args, send_args=None, timeout=5):
178 '''! Requête à l'arduino
179 @param opcode Code de l'opération
180 @param *args type des éléments reçu
181 @param send_args liste des paramètre qui vont avec l'envoi
182 @param timeout 5sec
183
184 @return Renvoie une liste data des données reçue
185 '''
186 data, data_size = ([], []) #Notre array
187 if(send_args==None):
188 self.order(opcode)# On envoit l'ordre pour avoir la réponse
189 else:
190 self.order(opcode, *send_args)
191
192 time.sleep(0.005)
193 startingtime = time.monotonic()
194 #Boucle while avec timeout
195 while True:
196 if(self.available()):#Tant que pas de réponse
197 for arg in args:#On cycle dans tout les valeurs excepté
198 datum, datum_size = self.read_buffer(arg) #On lit le buffer
199 data.append(datum)
200 data_size.append(datum_size)
201 break
202 else:#si on a pas de réponse, on la force mdr
203 if(send_args==None):
204 self.order(opcode)# On envoit l'ordre pour avoir la réponse
205 else:
206 self.order(opcode, *send_args)
207 time.sleep(0.1)
208 if(time.monotonic() - startingtime > timeout): raise TimeoutError
209
210 #On remet à 0 l'index RX
211 self.free_receiver()
212 return data

234 def read_buffer(self, obj_type):
235 '''! Lit le buffer RX
236 @param Type de l'obj
237 @return valeur
238 '''
239 data_size=-1
240 #On récupère la form de la donnée (liste ou valeur seule)
241 data_type = self.link.rx_obj(obj_type=STRING, start_pos=self.recSize, obj_byte_size=1)
242 self.recSize += txfer.STRUCT_FORMAT_LENGTHS[CHAR.format]
243 #Si valeur seule
244 if(data_type==SERIALTALKING_SINGLE_MAGIC):
245 data_size=1
246 data = self.link.rx_obj(obj_type=obj_type, start_pos=self.recSize)
247 self.recSize += txfer.STRUCT_FORMAT_LENGTHS[obj_type.format]
248
249 #Sinon si valeur multiple (liste)
250 elif(data_type==SERIALTALKING_MULTIPLE_MAGIC):
251 data_arr = []
252 #On récupère le nombre d'élément dans notre liste
253 data_size = self.link.rx_obj(obj_type=UCHAR, start_pos=self.recSize)
254 self.recSize += txfer.STRUCT_FORMAT_LENGTHS[UCHAR.format]
255
256 #~magie (juste on append un liste où les données sont les obj reçu)
257 for i in range(int(data_size)):
258
259 if(obj_type.format=="str"):
260 data_arr.append(self.link.rx_obj(obj_type=CHAR, start_pos=self.recSize))
261 self.recSize += 1
262 else:
263 data_arr.append(self.link.rx_obj(obj_type=obj_type, start_pos=self.recSize))
264 self.recSize += txfer.STRUCT_FORMAT_LENGTHS[obj_type.format]
265
266 if(obj_type.format=="str"):
267 data= b''.join(data_arr).decode()
268

270 return (-1,-1) #pas normal
