215 def _is_on_my_path(self, obstacle):
216 """
217 Check if the obstacle intersect the current path
218 """
219 if self.path is not None:
220 shape = obstacle.get_shape()
221 for i in range(len(self.path) - 1):
222 edge = self.path[i], self.path[i+1]
223 for i in range(len(shape)):
224 cutline = (shape[i], shape[(i + 1) % len(shape)])
225 if intersect(cutline, edge):
226 self.logger(WARNING, 'obstacle is on the path')
227 return True
228 else:
229 self.logger(INFO, 'cutline is not on the path')
230
231 self.logger(INFO, 'obstacle is not on the path')
232 return False
233 else:
234 return False

329 def move(self, destination, thresholds=(None, None)):
330 linpos_threshold, angpos_threshold = thresholds
331 default_linpos_threshold = 3
332 default_angpos_threshold = 0.1
333
334 # Set state to run
335 self.state = self.RUN
336
337 # Pathfinding
338 path_not_found = False
339 x_in, y_in, theta_in = self.wheeledbase.get_position()
340 x_sp, y_sp, theta_sp = destination
341
342 try:
343 self.path = self.roadmap.get_shortest_path(
344 (x_in, y_in), (x_sp, y_sp))
345 self.logger(INFO, 'follow path: [{}]'.format(', '.join(
346 '({0[0]:.0f}, {0[1]:.0f})'.format(waypoint) for waypoint in self.path)))
347 except RuntimeError:
348 path_not_found = True
349
350 # Return there is no path available
351 if path_not_found:
352 self.logger(ERROR, 'No path found !')
353 sleep(1)
354 return False
355
356 # Pure Pursuit configuration
357 if math.cos(math.atan2(self.path[1][1] - self.path[0][1], self.path[1][0] - self.path[0][0]) - theta_in) >= 0:
358 self.logger(INFO, 'Moving forward !')
359 self.direction = self.FORWARD
360 else:
361 self.logger(INFO, 'Moving backward !')
362 self.direction = self.BACKWARD
363
364 if math.hypot(self.path[1][0] - self.path[0][0], self.path[1][1] - self.path[0][1]) < 5 and theta_sp is not None:
365 finalangle = theta_sp - (self.direction - 1) * math.pi
366 else:
367 finalangle = None
368
369 self.wheeledbase.lookahead.set(150)
370 self.wheeledbase.lookaheadbis.set(150)
371 self.wheeledbase.max_linvel.set(300)
372 self.wheeledbase.max_angvel.set(6.0)
373 self.wheeledbase.linpos_threshold.set(
374 linpos_threshold or default_linpos_threshold)
375 self.wheeledbase.angpos_threshold.set(
376 angpos_threshold or default_angpos_threshold)
377
378 # Trajectory
379 self.wheeledbase.purepursuit(self.path, direction={
380 self.FORWARD: 'forward', self.BACKWARD: 'backward'}[self.direction], finalangle=finalangle)
381
382 # Wait until destination is reached
383 isarrived = False
384 blocked = False
385 while not isarrived:
386
387 try:
388 isarrived = self.wheeledbase.isarrived()
389 except RuntimeError:
390 self.logger(WARNING, 'Blocked while following path')
391 blocked = True
392
393 # If obstacle is on my path abort
394 if self.behaviour == self.BEHAVIOUR_STOPPING:
395 if self.abort.is_set():
396 timeout = time.time() + 5 # 5 seconds timemout
397 self.wheeledbase.stop()
398 while self.abort.is_set():
399 self.logger(INFO, 'Wait ...')
400 if time.time() > timeout:
401 self.logger(INFO, 'Timeout ! Retry to move')
402 return False
403 sleep(1)
404 self.logger(INFO, 'Continue ...')
405 return False
406 else:
407 if self.abort.is_set():
408 self.wheeledbase.stop()
409 self.abort.clear()
410 return False
411
412 if blocked:
413 self.logger(INFO, 'Go backward a little')
414 self.wheeledbase.set_velocities(-self.direction * 100, 0)
415 sleep(1)
416 self.logger(INFO, 'Resume path')
417 self.wheeledbase.purepursuit(
418 self.path, direction={self.FORWARD: 'forward', self.BACKWARD: 'backward'}[self.direction])
419 blocked = False
420
421 # Everything is fine
422 self.wheeledbase.linpos_threshold.set(default_linpos_threshold)
423 self.wheeledbase.angpos_threshold.set(default_angpos_threshold)
424
425 return True
