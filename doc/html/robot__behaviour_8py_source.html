202 def start(self):
203 """This method is the core of the robot. It allow the robot to go to each action and reach all action points
204 """
205 self.starttime = monotonic()
206 self.stop_event.clear()
207 self.logger(INFO, 'Start Behaviour')
208 try:
209 start = self.perform(self.start_procedure, timelimit=False)
210 while (self.timelimit is None or self.get_elapsed_time() < self.timelimit) and not self.stop_event.is_set():
211 decision = self.make_decision()
212 procedure, args, kwargs, (location, thresholds) = decision
213 if procedure is None:
214 self.logger(INFO, 'No decision')
215 sleep(1)
216 continue
217 if location is not None:
218 if location[2] is not None:
219 self.logger(
220 INFO, 'Goto: ({0[0]:.0f}, {0[1]:.0f}, {0[2]:.2f})'.format(location))
221 else:
222 self.logger(
223 INFO, 'Goto: ({0[0]:.0f}, {0[1]:.0f})'.format(location))
224 goto = self.perform(self.goto_procedure,
225 args=(location, thresholds))
226 success = self.get(goto)
227 else:
228 success = True
229 if success:
230 self.logger(INFO, "Perform Action")
231 action = self.perform(procedure, args=args, kwargs=kwargs)
232 self.get(action)
233 else:
234 self.logger(WARNING, 'Goto failed')
235 except:
236 self.whitelist.clear()
237 raise
238 finally:
239 self.stop()
240 self.get(start)
241 self.whitelist.add(id(current_thread()))
242 self.stop_procedure()
