251 def get_ressource(self, idx, name):
252 """Lock a shared mutex with the other robot
253
254 Args:
255 idx (int): The robot id
256 name (string): The ressource name
257
258 Raises:
259 RuntimeError: execution error
260
261 Returns:
262 bool: True or false
263 """
264 if not self.mutex.acquire(timeout=0.5):
265 return False
266
267 self.logger(INFO, "Ressource {} asking by {}".format(name, idx))
268 if not name in list(self.ressources.keys()):
269 self.logger(ERROR, "Unknown ressource !")
270 self.mutex.release()
271 raise RuntimeError("Unknown ressource !")
272
273 if self.ressources[name] != -1 and self.ressources[name] != idx:
274 try:
275 self.execute(_PING_OPCODE, id=self.ressources[name])
276 except (ConnectionError, TimeoutError, KeyError):
277 self.ressources[name] = idx
278 self.mutex.release()
279 return True
280 self.mutex.release()
281 self.logger(WARNING, "Rejected")
282 return False
283 else:
284 self.logger(INFO, "Mutex {} attributed to a {}".format(name, idx))
285 self.ressources[name] = idx
286 self.mutex.release()
287 return True

300 return
