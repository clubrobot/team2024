89 def join(self, timeout=None):
90 """
91 Wait until the thread terminates.
92
93 This blocks the calling thread until the thread whose join() method is
94 called terminates -- either normally or through an unhandled exception
95 or until the optional timeout occurs.
96
97 When the timeout argument is present and not None, it should be a
98 floating point number specifying a timeout for the operation in seconds
99 (or fractions thereof). As join() always returns None, you must call
100 isAlive() after join() to decide whether a timeout happened -- if the
101 thread is still alive, the join() call timed out.
102
103 When the timeout argument is not present or None, the operation will
104 block until the thread terminates.
105
106 A thread can be join()ed many times.
107
108 join() raises a RuntimeError if an attempt is made to join the current
109 thread as that would cause a deadlock. It is also an error to join() a
110 thread before it has been started and attempts to do so raises the same
111 exception.
112
113 """
114 self.logger(DEBUG, self.name, "join", timeout=timeout)
115 return th.Thread.join(self,timeout=timeout)
