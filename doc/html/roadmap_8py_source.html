11def intersect(A, B):
12 (xA, yA), (xA2, yA2) = A
13 (xB, yB), (xB2, yB2) = B
14 dxA, dyA = xA2 - xA, yA2 - yA
15 dxB, dyB = xB2 - xB, yB2 - yB
16 det = dyA * dxB - dxA * dyB
17 if det != 0:
18 tA = (-dyB * (xB - xA) + dxB * (yB - yA)) / det
19 tB = (-dyA * (xB - xA) + dxA * (yB - yA)) / det
20 return tA > 0 and tA < 1 and tB > 0 and tB < 1
