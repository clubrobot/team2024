9float VelocityController::genRampSetpoint(float stepSetpoint, float input, float rampSetpoint, float maxAcc, float maxDec, float timestep)
10{
11 // If we are above the desired setpoint (i.e. the ramp), we no longer try to follow it.
12 // Instead we generate a new ramp starting from our current position.
13 if ((input - rampSetpoint) * (stepSetpoint - rampSetpoint) > 0)
14 rampSetpoint = input;
15
16 // Do we have to accelerate or deccelerate to reach the desired setpoint?
17 if (input * (stepSetpoint - input) >= 0)
18 rampSetpoint += sign(stepSetpoint - input) * maxAcc * timestep;
19 else
20 rampSetpoint += sign(stepSetpoint - input) * maxDec * timestep;
21
22 // We clamp the ramp so that it never exceeds the real setpoint
23 if ((stepSetpoint - input) * (stepSetpoint - rampSetpoint) < 0) // TODO: test if this condition runs well when rampSetpoint is set to INFINITY
24 rampSetpoint = stepSetpoint;
25
26 return rampSetpoint;
27}

29void VelocityController::process(float timestep)
30{
31 // Save setpoints
32 const float stepLinVelSetpoint = m_linSetpoint;
33 const float stepAngVelSetpoint = m_angSetpoint;
34
35 // Compute new setpoints
36 m_rampLinVelSetpoint = genRampSetpoint(m_linSetpoint, m_linInput, m_rampLinVelSetpoint, m_maxLinAcc, m_maxLinDec, timestep);
37 m_rampAngVelSetpoint = genRampSetpoint(m_angSetpoint, m_angInput, m_rampAngVelSetpoint, m_maxAngAcc, m_maxAngDec, timestep);
38
39 // Do the engineering control
40 m_linSetpoint = m_rampLinVelSetpoint;
41 m_angSetpoint = m_rampAngVelSetpoint;
42 DifferentialController::process(timestep);
43
44 // Check for wheels abnormal spin and stop the controller accordingly
45 bool linVelSpin = (m_linVelOutput <= m_linPID->getMinOutput()) || (m_linVelOutput >= m_linPID->getMaxOutput());
46 bool angVelSpin = (m_angVelOutput <= m_angPID->getMinOutput()) || (m_angVelOutput >= m_angPID->getMaxOutput());
47 if (linVelSpin || angVelSpin)
48 {
49 bool abnormalSpin = (fabs(m_linInput) < 1) && (fabs(m_angInput) < 0.05);
50 if (abnormalSpin && m_spinShutdown)
51 {
52 m_leftWheel ->setVelocity(0);
53 m_rightWheel->setVelocity(0);
54 m_linSpinGoal = m_linSetpoint;
55 m_angSpinGoal = m_angSetpoint;
56 disable();
57 }
58 }
59
60 // Restore setpoints
61 m_linSetpoint = stepLinVelSetpoint;
62 m_angSetpoint = stepAngVelSetpoint;
